获取一个类的函数的引用本身：

    my $m = Str.^can('split')[0];
    > my $m = Str.^can('split')
    split split
    > $m.perl
    $(method split (Cool $: Any |) { #`(Method|63901144) ... },
      method split (Cool $: Any |) { #`(Method|63902600) ... })
    > $m.WHAT
    > say $m.WHAT
    (Parcel)
    > my @m = Str.^can('split')
    split split
    > @m.WHAT
    > say @m.WHAT
    (Array)
    > $m = [ Str.^can('split') ]
    split split
    > say $m.WHAT
    (Array)
